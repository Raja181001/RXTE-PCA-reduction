directories=$(find . -maxdepth 1 -type d -name "*result" -printf "%f\n")
curr=$(pwd)
for directory in $directories; do

    script_content="@${curr}/${directory}/rxte_nth_v1.xcm

    script_file="${curr}/${directory}/script1.txt"

    cd ${directory}

done
